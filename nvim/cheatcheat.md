# Nvim

I=insert mode beginning of the line
gi=insert last insert location

### insert mode only
ctrl+h=delete last char typed
ctrl+w=delete last word typed
ctrl+u=delete last line typed
ctrl+r"=put

u=undo
U=undo all changes to last line you changed
ctrl+r=redo

:sp=horizontal split
:vs=vertical split

gt=next tab
gT=previous tab

ctrl+v=block highlighting
o=front or back
O=font or back block highlighting

:{range}s/{pattern}/{replace}/{flags}
g=global
%=whole line

f=go to
t=go until
caps go backwards
;=next
,=previous

0=first char
^=first non-blank char
$=end of line
g_=last non black char

{=paragraph forward
}=paragraph back

J=join to next line
K=help for word

/=search forwards
?=search backwards
n=next match
N=previous match
*=/{word}
:noh=remove highlighting

ctrl+O=jump back
ctrl+I=jump forwards

%=match bracket

D=delete to end of line

C=change to end of line
cc=change complete line

.=repeat command
